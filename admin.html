STEP 2 — CREATE DATABASE TABLES
1
📋 Create Users Table Copy SQL
Run this in your Supabase SQL Editor (Dashboard → SQL Editor → New Query)
-- HATUA Users Table CREATE TABLE public.users ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, email TEXT UNIQUE NOT NULL, full_name TEXT, phone TEXT, avatar TEXT DEFAULT '🦁', plan TEXT DEFAULT 'trial', trial_start TIMESTAMPTZ DEFAULT now(), trial_end TIMESTAMPTZ DEFAULT (now() + interval '7 days'), points INTEGER DEFAULT 0, total_km NUMERIC DEFAULT 0, streak_days INTEGER DEFAULT 0, city TEXT DEFAULT 'Nairobi', is_banned BOOLEAN DEFAULT false, created_at TIMESTAMPTZ DEFAULT now(), last_active TIMESTAMPTZ DEFAULT now() );
2
💚 Create Payments Table
Records every M-Pesa transaction from Paybill 522522
-- HATUA M-Pesa Payments Table CREATE TABLE public.payments ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, user_id UUID REFERENCES public.users(id), phone TEXT NOT NULL, amount_kshs INTEGER NOT NULL, mpesa_ref TEXT UNIQUE, paybill TEXT DEFAULT '522522', plan TEXT NOT NULL, status TEXT DEFAULT 'pending', confirmed_at TIMESTAMPTZ, created_at TIMESTAMPTZ DEFAULT now() );
3
🛡 Create Security Logs Table
OWASP A09 — full audit trail of all security events
-- HATUA Security Audit Log CREATE TABLE public.security_logs ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, event_type TEXT NOT NULL, user_id UUID REFERENCES public.users(id), ip_address TEXT, user_agent TEXT, detail JSONB, severity TEXT DEFAULT 'info', created_at TIMESTAMPTZ DEFAULT now() ); -- Index for fast security dashboard queries CREATE INDEX idx_sec_logs_type ON security_logs(event_type); CREATE INDEX idx_sec_logs_time ON security_logs(created_at DESC);
4
🏃 Create Workouts Table
Stores every GPS session — powers AI route learning
-- HATUA Workouts / Sessions CREATE TABLE public.workouts ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, user_id UUID REFERENCES public.users(id), activity TEXT NOT NULL, route_name TEXT, distance_km NUMERIC, duration_s INTEGER, calories INTEGER, pts_earned INTEGER, avg_pace NUMERIC, mode TEXT DEFAULT 'outdoor', started_at TIMESTAMPTZ NOT NULL, ended_at TIMESTAMPTZ );
5
🔐 Enable Row-Level Security (RLS)
OWASP A01 — users can only see their own data
-- Enable RLS on all tables ALTER TABLE users ENABLE ROW LEVEL SECURITY; ALTER TABLE payments ENABLE ROW LEVEL SECURITY; ALTER TABLE workouts ENABLE ROW LEVEL SECURITY; ALTER TABLE security_logs ENABLE ROW LEVEL SECURITY; -- Users can only read/update their own row CREATE POLICY "users_own_data" ON users FOR ALL USING (auth.uid() = id); -- Users can only see their own payments CREATE POLICY "payments_own" ON payments FOR SELECT USING (auth.uid() = user_id); -- Admin service role bypasses RLS (admin panel uses service key)
6
⚙️ Update app.html with your Supabase keys
Add this at the top of your app.html <script> block, replacing the placeholder values:
// Supabase config — replace with your project values const SUPABASE_URL = 'https://YOUR_PROJECT.supabase.co'; const SUPABASE_KEY = 'YOUR_ANON_PUBLIC_KEY'; const sb = supabase.createClient(SUPABASE_URL, SUPABASE_KEY); // On app login — save to Supabase async function saveLogin(email) { await sb.from('security_logs').insert({ event_type: 'LOGIN_SUCCESS', detail: { email, ts: new Date().toISOString() } }); } // On workout end — save to Supabase async function saveWorkout(data) { await sb.from('workouts').insert(data); await sb.from('users') .update({ total_km: data.total_km, last_active: new Date() }) .eq('id', currentUserId); }
7
💚 M-Pesa Daraja Webhook (Optional but Recommended)
Auto-confirms payments when M-Pesa sends the callback. Create a Supabase Edge Function:
// supabase/functions/mpesa-webhook/index.ts import { createClient } from 'https://esm.sh/@supabase/supabase-js@2' Deno.serve(async (req) => { const body = await req.json(); const { Body: { stkCallback } } = body; if (stkCallback.ResultCode === 0) { const ref = stkCallback.CallbackMetadata.Item .find(i => i.Name === 'MpesaReceiptNumber')?.Value; const amount = stkCallback.CallbackMetadata.Item .find(i => i.Name === 'Amount')?.Value; const phone = stkCallback.CallbackMetadata.Item .find(i => i.Name === 'PhoneNumber')?.Value; // Update payment status in Supabase const sb = createClient(Deno.env.get('SUPABASE_URL')!, Deno.env.get('SERVICE_KEY')!); await sb.from('payments') .update({ status: 'confirmed', mpesa_ref: ref, confirmed_at: new Date() }) .eq('phone', phone).eq('status', 'pending'); return new Response('OK', { status: 200 }); } return new Response('Failed', { status: 400 }); });
